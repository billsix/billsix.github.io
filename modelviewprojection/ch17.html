Description¶

              ------------------------------ far z
              \             |              /
               \            |             /
                \ (x,z) *   |            /
                 \          |           /
                  \         |          /
                   \        |         /
                    \       |        /
                     \      |       /
                      \     |      /
                       ------------ near z
                       \    |    /
                        \   |   /
                         \  |  /
                          \ | /
                           \|/
      ----------------------*-(0,0)-------------------

because right angle and tan(theta) = tan(theta)
x / z = projX / nearZ
projX = x / z * nearZ

                       ----------- far z
                       |          |
                       |          |
        (x / z * nearZ,z) *       |   non-linear -- the transformation of x depends on its z value

149    def perspective(
150        self: Vertex, fov: float, aspectRatio: float, nearZ: float, farZ: float
151    ) -> Vertex:
152        # fov, field of view, is angle of y
153        # aspectRatio is xwidth / ywidth
154
155        top: float = -nearZ * math.tan(math.radians(fov) / 2.0)
156        right: float = top * aspectRatio
157
158        scaled_x: float = self.x / self.z * nearZ
159        scaled_y: float = self.y / self.z * nearZ
160        retangular_prism: Vertex = Vertex(scaled_x, scaled_y, self.z)
161
162        return retangular_prism.ortho(
163            left=-right, right=right, bottom=-top, top=top, near=nearZ, far=farZ
164        )
